/*
	Micromagnetic standard problem 4,
	with 3D discretization
*/

// geometry
	setgridsize(128, 32, 2)
	setcellsize(500e-9/128, 125e-9/32, 3e-9/2)

// material
	Msat = 800e3
	Aex = 13e-12
	m  = uniform(1, .1, 0)

// relax
	relax()
	TOL := 1e-5
	expectv("m", m.average(), vector(0.9669623970985413, 0.12529028952121735, 0), TOL)

// reversal
	alpha = 0.02
	B_ext = vector(-24.6E-3, 4.3E-3, 0)
	run(1e-9)
	expectv("m", m.average(), vector(-0.984630823135376, 0.12563535571098328, 0.04335508495569229), TOL)
