/*
	Micromagnetic standard problem 5
	as proposed by M. Najafi et al., JAP 105, 113914 (2009).
	Reference solution by mumax2.
*/

	setgridsize(32, 32, 4)
	setcellsize(100e-9/32, 100e-9/32, 10e-9/4)

	Msat  = 800e3
	Aex   = 13e-12
	m     = vortex(1, 1)
	alpha = 3
 
	run(1e-9) // relax

	alpha = 0.1
	xi    = 0.05
	JPol  = vector(1e12, 0, 0)

	run(1e-9) // run

	m_ := average(m)
	expect("mx", m_[0], -0.23479773, 1e-5)
	expect("my", m_[1], -0.09453578, 1e-5)
	expect("mz", m_[2],  0.02296375, 1e-5)
